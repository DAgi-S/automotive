Automotive Database Structure

1. User Management

tbl_user

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
name varchar(255) NOT NULL User's full name
email varchar(255) NOT NULL, UNIQUE User's email address
password varchar(255) NOT NULL Hashed password
role varchar(20) NULL User's role
phonenum varchar(20) NOT NULL Phone number
created_at timestamp NOT NULL Account creation timestamp

tbl_worker

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
full_name varchar(255) NOT NULL Worker's full name
username varchar(100) NOT NULL, UNIQUE Login username
password varchar(255) NOT NULL Hashed password
position enum('worker') NOT NULL Worker position
created_at timestamp NOT NULL Account creation timestamp

tbl_admin

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
full_name varchar(255) NOT NULL Admin's full name
username varchar(100) NOT NULL, UNIQUE Login username
password varchar(255) NULL Hashed password
position enum('admin','superadmin') NULL Admin position
created_at timestamp NOT NULL Account creation timestamp

2. Vehicle Management

tbl_vehicles

Column Data Type Constraints Description
vehicle_id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
user_id int(11) NULL, FOREIGN KEY Reference to tbl_user
make varchar(100) NULL Vehicle make
model varchar(100) NULL Vehicle model
year int(11) NULL Vehicle year
license_plate varchar(20) NULL License plate number
created_at timestamp NOT NULL Record creation timestamp

car_brands

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
brand_name varchar(100) NOT NULL, UNIQUE Name of the car brand
created_at timestamp NOT NULL Record creation timestamp

car_models

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
brand_id int(11) NOT NULL, FOREIGN KEY Reference to car_brands
model_name varchar(100) NOT NULL Name of the car model
year_from int(4) NULL Starting year of the model
year_to int(4) NULL Ending year of the model
created_at timestamp NOT NULL Record creation timestamp

tbl_info

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
userid int(11) NULL Reference to user
car_brand varchar(100) NULL Car brand
car_year varchar(10) NULL Car year
car_model varchar(100) NULL Car model
plate_number varchar(20) NULL License plate number
service_date varchar(20) NULL Service date
mile_age varchar(20) NULL Vehicle mileage
created_at timestamp NOT NULL Record creation timestamp

3. Service Management

tbl_services

Column Data Type Constraints Description
service_id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
service_name varchar(255) NOT NULL Name of the service
description text NULL Service description
price decimal(10,2) NULL Service price
status enum('active','inactive') NULL Service status
created_at timestamp NOT NULL Record creation timestamp

tbl_service_history

Column Data Type Constraints Description
history_id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
vehicle_id int(11) NULL, FOREIGN KEY Reference to tbl_vehicles
service_id int(11) NULL, FOREIGN KEY Reference to tbl_services
service_date date NULL Date of service
notes text NULL Service notes
cost decimal(10,2) NULL Service cost
created_at timestamp NOT NULL Record creation timestamp

tbl_appointments

Column Data Type Constraints Description
appointment_id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
user_id int(11) NULL, FOREIGN KEY Reference to tbl_user
service_id int(11) NULL, FOREIGN KEY Reference to tbl_services
appointment_date date NULL Appointment date
appointment_time time NULL Appointment time
status enum('pending','confirmed','completed','cancelled') NULL Appointment status
notes text NULL Appointment notes
worker_id int(11) NULL, FOREIGN KEY Reference to tbl_worker
created_at timestamp NOT NULL Record creation timestamp

4. Product Management

tbl_products

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
name varchar(255) NOT NULL Product name
description text NULL Product description
price decimal(10,2) NOT NULL Product price
category_id int(11) NULL, FOREIGN KEY Reference to category
stock int(11) NULL Available stock
status enum('active','inactive') NULL Product status
created_at timestamp NOT NULL Record creation timestamp

tbl_categories

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
name varchar(255) NOT NULL Category name
description text NULL Category description
status enum('active','inactive') NULL Category status
created_at timestamp NOT NULL Record creation timestamp
updated_at timestamp NOT NULL Last update timestamp

tbl_cart

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
user_id int(11) NULL, FOREIGN KEY Reference to tbl_user
product_id int(11) NULL, FOREIGN KEY Reference to tbl_products
quantity int(11) NULL Quantity in cart
added_date timestamp NOT NULL Item addition timestamp

tbl_wishlist

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
user_id int(11) NULL, FOREIGN KEY Reference to tbl_user
product_id int(11) NULL, FOREIGN KEY Reference to tbl_products
added_date timestamp NOT NULL Item addition timestamp

5. Content Management

tbl_blog

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
title varchar(255) NOT NULL Blog post title
writer varchar(100) NOT NULL Author name
content text NULL Blog content
image_url varchar(255) NULL Featured image URL
status varchar(20) NULL Post status
date datetime NULL Publication date

tbl_blog_comments

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
blog_id int(11) NOT NULL, FOREIGN KEY Reference to tbl_blog
comment text NOT NULL Comment content
parent_id int(11) NULL, FOREIGN KEY Reference to parent comment
created_at datetime NULL Comment creation timestamp

tbl_tiktok_videos

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
video_url varchar(255) NOT NULL TikTok video URL
video_id varchar(100) NOT NULL TikTok video ID
title varchar(255) NULL Video title
description text NULL Video description
created_at timestamp NOT NULL Record creation timestamp

6. System Management

permissions

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
permission_name varchar(100) NOT NULL, UNIQUE Name of the permission
permission_description text NULL Permission description
created_at timestamp NOT NULL Record creation timestamp

role_permissions

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
role_id int(11) NOT NULL, FOREIGN KEY Reference to roles
permission_id int(11) NOT NULL, FOREIGN KEY Reference to permissions
created_at timestamp NOT NULL Record creation timestamp

activity_log

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
user_id int(11) NULL User who performed the action
action varchar(255) NOT NULL Action performed
details text NULL Action details
created_at timestamp NOT NULL Activity timestamp

7. Communication

tbl_message

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
name varchar(255) NULL Sender's name
email varchar(255) NULL Sender's email
subject varchar(255) NULL Message subject
status enum('read','unread') NULL Message status
created_at timestamp NOT NULL Message creation timestamp

tbl_notifications

Column Data Type Constraints Description
id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
type varchar(50) NULL Notification type
reference_id int(11) NULL Reference to related record
message text NULL Notification message
recipient_id int(11) NULL, FOREIGN KEY Recipient user ID
is_read tinyint(1) NULL Read status
created_at timestamp NOT NULL Notification creation timestamp

8. Reporting System

tbl_reports

Column Data Type Constraints Description
report_id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
report_type varchar(50) NOT NULL Type of report (sales, services, etc.)
start_date date NOT NULL Report start date
end_date date NOT NULL Report end date
format varchar(10) NOT NULL Report format (html, pdf, csv, excel)
file_path varchar(255) NULL Path to stored report file
admin_id int(11) NOT NULL, FOREIGN KEY Reference to tbl_admin
created_at timestamp NOT NULL Report generation timestamp

tbl_report_parameters

Column Data Type Constraints Description
parameter_id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
report_id int(11) NOT NULL, FOREIGN KEY Reference to tbl_reports
parameter_name varchar(50) NOT NULL Name of the parameter
parameter_value text NULL Value of the parameter
created_at timestamp NOT NULL Parameter creation timestamp

tbl_report_schedules

Column Data Type Constraints Description
schedule_id int(11) PRIMARY KEY, AUTO_INCREMENT Unique identifier
report_type varchar(50) NOT NULL Type of report to generate
schedule_name varchar(100) NOT NULL Name of the schedule
frequency enum('daily','weekly','monthly','yearly') NOT NULL Report generation frequency
format varchar(10) NOT NULL Report format
recipients text NULL Email recipients (comma-separated)
last_run timestamp NULL Last execution timestamp
next_run timestamp NOT NULL Next execution timestamp
status enum('active','inactive') NOT NULL Schedule status
created_by int(11) NOT NULL, FOREIGN KEY Reference to tbl_admin
created_at timestamp NOT NULL Schedule creation timestamp
Note: This documentation represents the complete database structure. The database uses proper indexing, foreign key relationships, and timestamp tracking for audit purposes. All tables follow consistent naming conventions and include appropriate constraints for data integrity. The reporting system supports multiple report types, formats, and scheduled report generation.
